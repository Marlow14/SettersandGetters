public class Setters_And_Getters();

1) The size of a disk file

private double discSize;

public double getDiscSize() {
	return discSize;
}

public void setDiscSize(double discSize) {
	this.discSize = discSize;
}

2) The number of acres in a plot of land

private int acresOfLand;

public int getAcresOfLand() {
	return acresOfLand;
}

public void setAcresOfLand(int acresOfLand) {
	this.acresOfLand = acresOfLand;
}

private String fileName;

public String getFileName() {

public void setFileName(String fileName) {

}
